char*StrCpy(char*src)
{
//计算src的长度

int len=0;
for(char*p=src;*p!=NULL;p++)
len++；

//申请len+1长度的内存空间
char*dst=(char*)malloc(sizeof(char)*(len+1));

//字符复刻
for(int i=0;i<len;i++)
dst[i]=src[i];

dst[len]='\0';

return dst;
}
